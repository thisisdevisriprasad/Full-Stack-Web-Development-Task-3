  <script>
    // Toggle Between Forms
    function toggleForms() {
      const login = document.getElementById('loginForm');
      const register = document.getElementById('registerForm');
      if (login.style.display === 'block') {
        login.style.display = 'none';
        register.style.display = 'block';
      } else {
        register.style.display = 'none';
        login.style.display = 'block';
      }
    }

    // Password Toggle
    const toggleLoginPass = document.getElementById('toggleLoginPass');
    const loginPassword = document.getElementById('loginPassword');
    toggleLoginPass.addEventListener('click', () => {
      const type = loginPassword.getAttribute('type') === 'password' ? 'text' : 'password';
      loginPassword.setAttribute('type', type);
      toggleLoginPass.classList.toggle('fa-eye-slash');
    });

    const toggleRegPass = document.getElementById('toggleRegPass');
    const regPassword = document.getElementById('regPassword');
    toggleRegPass.addEventListener('click', () => {
      const type = regPassword.getAttribute('type') === 'password' ? 'text' : 'password';
      regPassword.setAttribute('type', type);
      toggleRegPass.classList.toggle('fa-eye-slash');
    });

  
// Login Validation
document.getElementById('loginFormElement').addEventListener('submit', function (e) {
  e.preventDefault();
  const email = loginEmail.value.trim();
  const pass = loginPassword.value.trim();

  let valid = true;
  if (!email || !email.includes('@')) {
    loginEmail.classList.add('is-invalid'); valid = false;
  } else { loginEmail.classList.remove('is-invalid'); }
  
  if (!pass) {
    loginPassword.classList.add('is-invalid'); valid = false;
  } else { loginPassword.classList.remove('is-invalid'); }

  if (!valid) return;

  // Send login request to PHP
  fetch('logic.php', {
    method: 'POST',
    headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
    body: `email=${encodeURIComponent(email)}&password=${encodeURIComponent(pass)}`
  })
  .then(res => res.text())
  .then(response => {
    console.log(response); // Debug line, check console for actual PHP output
    if(response.trim() === 'success'){
      alert('Login Successful!');
      this.reset();
    } else if(response.trim() === 'invalid'){
      alert('Incorrect password.');
    } else if(response.trim() === 'notfound'){
      alert('User not found. Please register.');
    } else {
      alert('Error: ' + response);
    }
  })
  .catch(err => console.error(err));
});

    // Register Validation
document.getElementById('registerFormElement').addEventListener('submit', function (e) {
  e.preventDefault();
  const name = regName.value.trim();
  const email = regEmail.value.trim();
  const pass = regPassword.value.trim();
  const confirm = regConfirm.value.trim();

  let valid = true;

  if (name === '') { regName.classList.add('is-invalid'); valid = false; } else { regName.classList.remove('is-invalid'); }
  if (!email.includes('@')) { regEmail.classList.add('is-invalid'); valid = false; } else { regEmail.classList.remove('is-invalid'); }
  if (pass.length < 6) { regPassword.classList.add('is-invalid'); valid = false; } else { regPassword.classList.remove('is-invalid'); }
  if (confirm !== pass || confirm === '') { regConfirm.classList.add('is-invalid'); valid = false; } else { regConfirm.classList.remove('is-invalid'); }

  if (!valid) return;

  // Send registration request to PHP
  fetch('register.php', {
    method: 'POST',
    headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
    body: `fullname=${encodeURIComponent(name)}&email=${encodeURIComponent(email)}&password=${encodeURIComponent(pass)}`
  })
  .then(res => res.text())
  .then(response => {
    alert(response);
    if(response.includes('success')){
      this.reset();
      toggleForms();
    }
  })
  .catch(err => console.error(err));
});
